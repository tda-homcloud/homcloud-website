# PH0のコンポーネントを表示する関数
# ax は matplotlib の Axes オブジェクトで，
# pair は homcloud の birth-death pair オブジェクト
def show_ph0_component(ax, pair):
    # birth time と death time を タイトルに表示
    ax.set_title(f"{pair.birth_time()}, {pair.death_time()}")
    # 縦横比を揃える
    ax.set_aspect("equal")
    # 入力ポイントクラウドを黒で描画
    ax.scatter(pointcloud[:, 0], pointcloud[:, 1], s=2, c="k")
    # birth-death pairからコンポーネントを取り出す
    components = pair.ph0_components()
    def show(vertices, color):
        ax.scatter(vertices[:, 0], vertices[:, 1], s=4, c=color)
    # ここで birth-component と elder_componentを描画
    # alpha filtration で得られる PD に関しては，この2つの塊が
    # 円が大きくなってくっついたときに death となる
    # この2つのコンポーネント間の最短距離をrとすると，
    # death = r * r / 4 である
    # birth と elder と名前が付いているが，この場合にはどちらがどちらという差異がない
    # elder のほうが大きいコンポーネントになりがち，という傾向はある
    show(components.birth_component, "r")
    show(components.elder_component, "g")

黒: ポイントクラウドデータ全体 赤: そのbirth-death pairに対応するクラスタ 緑: そのbirth-death pairの「親」となるクラスタ
