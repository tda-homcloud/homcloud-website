def show_ph0_component(ax, pair):
    # birth time  death time  
    ax.set_title(f"{pair.birth_time()}, {pair.death_time()}")
    # 
    ax.set_aspect("equal")
    # 
    ax.scatter(pointcloud[:, 0], pointcloud[:, 1], s=2, c="k")
    # birth-death pair
    components = pair.ph0_components()
    def show(vertices, color):
        ax.scatter(vertices[:, 0], vertices[:, 1], s=4, c=color)
    #  birth-component  elder_component
    # alpha filtration  PD 2
    #  death 
    # 2r
    # death = r * r / 4 
    # birth  elder 
    # elder 
    show(components.birth_component, "r")
    show(components.elder_component, "g")
