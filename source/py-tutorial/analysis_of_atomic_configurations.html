import re

with open("sio2-cart.chem3d") as infile:
    with open("sio2-cart.xyz", "w") as outfile:
        outfile.write(infile.readline())
        outfile.write("SiO2\n")
        for line in infile:
            atom, index, x, y, z = re.split(r"\s+", line.strip())
            outfile.write(f"{atom} {x} {y} {z}\n")
